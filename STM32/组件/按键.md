# 常规按键
使用阻塞延时

```
#define WK_NUM 0
#define KEY0_NUM 1

#define WK_LEVEL 读取WK端口点屏的函数
#define KEY0_LEVEL 读取KEY0端口点屏的函数

uint8_t KEY_Scan(void)
{
	static uint8_t key_up = 1;
	uint8_t key_num = 0xff;

	if(key_up && (WK_LEVEL || !KEY0_LEVEL))
	{
		delay_ms(20);
		key_up = 0;
		if(WK_LEVEL) key_num = WK_NUM;
		else if(!KEY0_LEVEL) key_num = KEY0_NUM;
	}
	else if((!WK_LEVEL) && KEY0_LEVEL) // 得松开才能有结果
	{
		key_up = 1;
	}

	return key_num;
}
```
